// oc-buy-now
window.oc.events.fire('webshop:add-to-cart', {
  id: 'product123',
  amount: 1
});

// oc-basket
window.oc.events.on("webshop:add-to-cart", (_eventDetails, eventData) => {
  const { id, amount } = eventData;
  handleAddToCart(id, amount);
});
